  }, 2000);

```

---

```js

  setTimeout(() => {
    resolve('Data received successfully');
  }, 2000);
});
```

---

```js

  setTimeout(() => {
    resolve('Data received successfully');
  }, 2000);
});
```

  setTimeout(() => {
    resolve('Data received successfully');
  }, 2000);
});
```
